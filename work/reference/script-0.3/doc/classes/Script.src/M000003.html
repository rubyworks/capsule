  def require(feature)
    unless @loaded_features[feature]
      @loaded_features[feature] = true
      file = File.join(@dir, feature)
      file += ".rb" unless /\.rb$/ =~ file
      load_in_module(file)
    end
  rescue MissingFile
    @loaded_features[feature] = false
    super
  end
